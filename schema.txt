TABLE Users {
    id INTEGER
    firstName varchar
    lastName varchar
    email varchar
    previewImage REFERENCES images.id
}
TABLE GROUPS {4
    id INTEGER
    organizerId INTEGER REFERENCES users.id
    name varchar
    about blob
    type    string
    private boolean
    city    string
    state   string
    previewImage REFERENCES images.id
}

TABLE VENUES {
    id INTEGER
    groupId INTEGER references groups.id
    address varchar
    city varchar
    state varchar
    lat float (9,6)
    lng float (9,6)
}

TABLE EVENTS {
    id INTEGER
    groupId INTEGER references groups.id
    venueId INTEGER references venues.id
    name varchar
    type varchar
    startDate DATE
    endDate DATE
    numAttending INTEGER
}

TABLE MEMBERSHIPS {
    id INTEGER
    groupId INTEGER references groups.id
    memberId INTEGER references members.id
    status varchar
}

TABLE ATTENDEES {
    id INTEGER
    eventId INTEGER REFERENCES events.id
    userId INTEGER references users.id
    status varchar
}

TABLE IMAGES {
    id INTEGER
    image URL
}
